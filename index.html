          <script type="text/javascript">
            var form = document.getElementById("cb_web_form");
            var ticketQuantity = document.getElementById("ticketQuantity");
            var amount = document.getElementById("AMOUNT_EVT");
            var total = document.getElementById("AMOUNT_PAID");

            function getIntValue(field) {
              if (field == null) return 0;
              var value = parseFloat(field.value);
              if (isNaN(value)) return 0;
              return value;
            }

            function calculatePaymentTotal() {
              total.value = getIntValue(ticketQuantity);
              total.value = total.value * {{formPrice}};
              amount.value = total.value;
              return true;
            }
            form.onsubmit = ticketQuantity.onchange = calculatePaymentTotal;
          </script>
